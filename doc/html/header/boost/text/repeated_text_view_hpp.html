Header <boost/text/repeated_text_view.hpp>

namespace boost {
  namespace text {
    struct repeated_text_view;
    constexpr repeated_text_view repeat(text_view, std::ptrdiff_t);
    constexpr bool operator==(repeated_text_view, repeated_text_view);
    constexpr bool operator!=(repeated_text_view, repeated_text_view);
    constexpr bool operator<(repeated_text_view, repeated_text_view);
    constexpr bool operator<=(repeated_text_view, repeated_text_view);
    constexpr bool operator>(repeated_text_view, repeated_text_view);
    constexpr bool operator>=(repeated_text_view, repeated_text_view);
    constexpr repeated_text_view::iterator begin(repeated_text_view rtv);
    constexpr repeated_text_view::iterator end(repeated_text_view rtv);
    constexpr repeated_text_view::reverse_iterator 
    rbegin(repeated_text_view rtv);
    constexpr repeated_text_view::reverse_iterator 
    rend(repeated_text_view rtv);
  }
}
