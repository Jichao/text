Header <boost/text/rope.hpp>

namespace boost {
  namespace text {
    struct rope;
    rope const & checked_encoding(rope const &);
    rope && checked_encoding(rope &&);
    rope operator+(rope, rope);
    rope operator+(rope, rope_view);
    rope operator+(rope_view, rope);
    rope operator+(rope, text &&);
    rope operator+(text &&, rope);
    rope_view::iterator begin(rope_view rv);
    rope_view::iterator end(rope_view rv);
    rope_view::reverse_iterator rbegin(rope_view rv);
    rope_view::reverse_iterator rend(rope_view rv);
    std::ostream & operator<<(std::ostream &, rope_view);
    rope operator+(rope_view, rope_view);
  }
}
