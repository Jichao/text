Header <boost/text/text.hpp>

namespace boost {
  namespace text {
    struct text;
    text::iterator begin(text & t);
    text::iterator end(text & t);
    text::const_iterator begin(text const & t);
    text::const_iterator end(text const & t);
    text::const_iterator cbegin(text const & t);
    text::const_iterator cend(text const & t);
    text::reverse_iterator rbegin(text & t);
    text::reverse_iterator rend(text & t);
    text::const_reverse_iterator rbegin(text const & t);
    text::const_reverse_iterator rend(text const & t);
    text::const_reverse_iterator crbegin(text const & t);
    text::const_reverse_iterator crend(text const & t);
    bool operator==(text const & lhs, char const * rhs);
    bool operator==(char const * lhs, text const & rhs);
    bool operator!=(text const & lhs, char const * rhs);
    bool operator!=(char const * lhs, text const & rhs);
    bool operator==(text const & lhs, text_view rhs);
    bool operator==(text_view lhs, text const & rhs);
    bool operator!=(text const & lhs, text_view rhs);
    bool operator!=(text_view lhs, text const & rhs);
    bool operator==(text const & lhs, text const & rhs);
    bool operator!=(text const & lhs, text const & rhs);
    text operator+(text, text const &);
    text operator+(text, char const *);
    text operator+(char const *, text const &);
    text operator+(text, text_view);
    text operator+(text_view, text const &);
    text operator+(text, repeated_string_view);
    text operator+(repeated_string_view, text const &);
    template<typename CharRange> text operator+(text, CharRange const &);
    template<typename CharRange> 
      text operator+(CharRange const &, text const &);
    namespace literals {
      text operator""_t(char const *, std::size_t);
    }
  }
}
