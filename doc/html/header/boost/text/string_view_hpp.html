Header <boost/text/string_view.hpp>

namespace boost {
  namespace text {
    struct string_view;
    constexpr bool operator==(string_view, string_view);
    constexpr bool operator==(char const *, string_view);
    constexpr bool operator==(string_view, char const *);
    constexpr bool operator!=(string_view, string_view);
    constexpr bool operator!=(char const *, string_view);
    constexpr bool operator!=(string_view, char const *);
    constexpr bool operator<(string_view, string_view);
    constexpr bool operator<(char const *, string_view);
    constexpr bool operator<(string_view, char const *);
    constexpr bool operator<=(string_view, string_view);
    constexpr bool operator<=(char const *, string_view);
    constexpr bool operator<=(string_view, char const *);
    constexpr bool operator>(string_view, string_view);
    constexpr bool operator>(char const *, string_view);
    constexpr bool operator>(string_view, char const *);
    constexpr bool operator>=(string_view, string_view);
    constexpr bool operator>=(char const *, string_view);
    constexpr bool operator>=(string_view, char const *);
    constexpr string_view::iterator begin(string_view sv);
    constexpr string_view::iterator end(string_view sv);
    constexpr string_view::iterator cbegin(string_view sv);
    constexpr string_view::iterator cend(string_view sv);
    constexpr string_view::reverse_iterator rbegin(string_view sv);
    constexpr string_view::reverse_iterator rend(string_view sv);
    constexpr string_view::reverse_iterator crbegin(string_view sv);
    constexpr string_view::reverse_iterator crend(string_view sv);
    int operator+(string_view lhs, char const * rhs);
    int operator+(char const * lhs, string_view rhs);
    namespace literals {
      constexpr string_view operator""_sv(char const *, std::size_t);
    }
  }
}
