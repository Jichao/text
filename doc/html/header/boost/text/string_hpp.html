Header <boost/text/string.hpp>

namespace boost {
  namespace text {
    struct string;
    string::iterator begin(string & s);
    string::iterator end(string & s);
    string::const_iterator begin(string const & s);
    string::const_iterator end(string const & s);
    string::const_iterator cbegin(string const & s);
    string::const_iterator cend(string const & s);
    string::reverse_iterator rbegin(string & s);
    string::reverse_iterator rend(string & s);
    string::const_reverse_iterator rbegin(string const & s);
    string::const_reverse_iterator rend(string const & s);
    string::const_reverse_iterator crbegin(string const & s);
    string::const_reverse_iterator crend(string const & s);
    bool operator==(char const * lhs, string const & rhs);
    bool operator==(string const & lhs, char const * rhs);
    bool operator!=(char const * lhs, string const & rhs);
    bool operator!=(string const & lhs, char const * rhs);
    bool operator<(char const * lhs, string const & rhs);
    bool operator<(string const & lhs, char const * rhs);
    bool operator<=(char const * lhs, string const & rhs);
    bool operator<=(string const & lhs, char const * rhs);
    bool operator>(char const * lhs, string const & rhs);
    bool operator>(string const & lhs, char const * rhs);
    bool operator>=(char const * lhs, string const & rhs);
    bool operator>=(string const & lhs, char const * rhs);
    template<typename CharRange> 
      auto operator==(CharRange const & lhs, string const & rhs);
    template<typename CharRange> 
      auto operator==(string const & lhs, CharRange const & rhs);
    template<typename CharRange> 
      auto operator!=(CharRange const & lhs, string const & rhs);
    template<typename CharRange> 
      auto operator!=(string const & lhs, CharRange const & rhs);
    template<typename CharRange> 
      auto operator<(CharRange const & lhs, string const & rhs);
    template<typename CharRange> 
      auto operator<(string const & lhs, CharRange const & rhs);
    template<typename CharRange> 
      auto operator<=(CharRange const & lhs, string const & rhs);
    template<typename CharRange> 
      auto operator<=(string const & lhs, CharRange const & rhs);
    template<typename CharRange> 
      auto operator>(CharRange const & lhs, string const & rhs);
    template<typename CharRange> 
      auto operator>(string const & lhs, CharRange const & rhs);
    template<typename CharRange> 
      auto operator>=(CharRange const & lhs, string const & rhs);
    template<typename CharRange> 
      auto operator>=(string const & lhs, CharRange const & rhs);
    string operator+(string, char);
    string operator+(string, char const *);
    string operator+(char const *, string);
    string operator+(string const &, string_view);
    string operator+(string &&, string_view);
    string operator+(string const &, string const &);
    string operator+(string &&, string const &);
    string operator+(string const &, string &&);
    string operator+(string &&, string &&);
    string operator+(string const &, repeated_string_view);
    string operator+(string &&, repeated_string_view);
    string operator+(string_view, string const &);
    string operator+(repeated_string_view, string const &);
    template<typename CharRange> string operator+(string, CharRange const &);
    template<typename CharRange> 
      string operator+(CharRange const &, string const &);
    namespace literals {
      string operator""_s(char const *, std::size_t);
    }
  }
}
