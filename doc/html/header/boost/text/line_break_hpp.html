  namespace text {
    template<typename CPIter, typename Sentinel = CPIter> 
      struct line_break_cp_range;
    template<typename CPIter> struct line_break_grapheme_range;
    template<typename CPIter> struct line_break_result;

    enum line_property;
    line_property line_prop(uint32_t);
    template<typename CPIter, typename Sentinel> 
      auto operator==(line_break_result< CPIter > result, Sentinel s);
    template<typename CPIter> 
      auto operator==(CPIter it, line_break_result< CPIter > result);
    template<typename CPIter, typename Sentinel> 
      auto operator!=(line_break_result< CPIter > result, Sentinel s);
    template<typename CPIter, typename Sentinel> 
      auto operator!=(CPIter it, line_break_result< CPIter > result);
    template<typename CPIter, typename Sentinel> 
      CPIter prev_hard_line_break(CPIter, CPIter, Sentinel);
    template<typename CPIter, typename Sentinel> 
      line_break_result< CPIter > 
      prev_allowed_line_break(CPIter, CPIter, Sentinel);
    template<typename CPIter, typename Sentinel> 
      CPIter next_hard_line_break(CPIter, Sentinel);
    template<typename CPIter, typename Sentinel> 
      CPIter next_allowed_line_break(CPIter, Sentinel);
    template<typename CPRange, typename CPIter> 
      unspecified prev_hard_line_break(CPRange &, CPIter);
    template<typename GraphemeRange, typename GraphemeIter> 
      unspecified prev_hard_line_break(GraphemeRange const &, GraphemeIter);
    template<typename CPRange, typename CPIter> 
      unspecified next_hard_line_break(CPRange &, CPIter);
    template<typename GraphemeRange, typename GraphemeIter> 
      unspecified next_hard_line_break(GraphemeRange const &, GraphemeIter);
    template<typename CPRange, typename CPIter> 
      unspecified prev_allowed_line_break(CPRange &, CPIter);
    template<typename GraphemeRange, typename GraphemeIter> 
      unspecified prev_allowed_line_break(GraphemeRange const &, GraphemeIter);
    template<typename CPRange, typename CPIter> 
      unspecified next_allowed_line_break(CPRange &, CPIter);
    template<typename GraphemeRange, typename GraphemeIter> 
      unspecified next_allowed_line_break(GraphemeRange const &, GraphemeIter);
    template<typename CPIter, typename Sentinel> 
      cp_range< CPIter > line(CPIter, CPIter, Sentinel);
    template<typename CPRange, typename CPIter> 
      unspecified line(CPRange &, CPIter);
    template<typename GraphemeRange, typename GraphemeIter> 
      unspecified line(GraphemeRange const &, GraphemeIter);
    template<typename CPIter, typename Sentinel> 
      unspecified lines(CPIter, Sentinel);
    template<typename CPRange> unspecified lines(CPRange &);
    template<typename GraphemeRange> unspecified lines(GraphemeRange const &);
    template<typename CPIter> unspecified reversed_lines(CPIter, CPIter);
    template<typename CPRange> unspecified reversed_lines(CPRange &);
    template<typename GraphemeRange> 
      unspecified reversed_lines(GraphemeRange const &);
    template<typename CPIter, typename Sentinel, typename Extent, 
             typename CPExtentFunc> 
      unspecified lines(CPIter, Sentinel, Extent, CPExtentFunc, bool = true);
    template<typename CPRange, typename Extent, typename CPExtentFunc> 
      unspecified lines(CPRange &, Extent, CPExtentFunc, bool = true);
    template<typename GraphemeRange, typename Extent, typename CPExtentFunc> 
      unspecified lines(GraphemeRange const &, Extent, CPExtentFunc, 
                        bool = true);
    template<typename CPIter, typename Sentinel> 
      line_break_cp_range< CPIter > allowed_line(CPIter, CPIter, Sentinel);
    template<typename CPRange, typename CPIter> 
      unspecified allowed_line(CPRange &, CPIter);
    template<typename GraphemeRange, typename GraphemeIter> 
      unspecified allowed_line(GraphemeRange const &, GraphemeIter);
    template<typename CPIter, typename Sentinel> 
      unspecified allowed_lines(CPIter, Sentinel);
    template<typename CPRange> unspecified allowed_lines(CPRange &);
    template<typename GraphemeRange> 
      unspecified allowed_lines(GraphemeRange const &);
    template<typename CPIter> 
      unspecified reversed_allowed_lines(CPIter, CPIter);

      unspecified reversed_allowed_lines(GraphemeRange const &);
