namespace boost {
  namespace text {
    enum line_property;
    line_property line_prop(uint32_t);
    template<typename CPIter, typename Sentinel> 
      CPIter prev_hard_line_break(CPIter, CPIter, Sentinel);
    template<typename CPIter, typename Sentinel> 
      CPIter prev_possible_line_break(CPIter, CPIter, Sentinel);
    template<typename CPIter, typename Sentinel> 
      CPIter next_hard_line_break(CPIter, Sentinel);
    template<typename CPIter, typename Sentinel> 
      CPIter next_possible_line_break(CPIter, Sentinel);
    template<typename CPRange, typename CPIter> 
      auto prev_hard_line_break(CPRange &, CPIter);
    template<typename CPRange> auto next_hard_line_break(CPRange &);
    template<typename CPRange, typename CPIter> 
      auto prev_possible_line_break(CPRange &, CPIter);
    template<typename CPRange> auto next_possible_line_break(CPRange &);
    template<typename CPIter, typename Sentinel> 
      cp_range< CPIter > line(CPIter, CPIter, CPIter);
    template<typename CPRange, typename CPIter> auto line(CPRange &, CPIter);
    template<typename CPIter, typename Sentinel> 
      unspecified lines(CPIter, CPIter);
    template<typename CPRange> auto lines(CPRange &);
    template<typename CPIter, typename Sentinel> 
      cp_range< CPIter > possible_line(CPIter, CPIter, Sentinel);
    template<typename CPRange, typename CPIter> 
      auto possible_line(CPRange &, CPIter);
    template<typename CPIter, typename Sentinel> 
      unspecified possible_lines(CPIter, Sentinel);
    template<typename CPRange> auto possible_lines(CPRange &);
  }
}
