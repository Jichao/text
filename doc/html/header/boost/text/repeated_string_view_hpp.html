Header <boost/text/repeated_string_view.hpp>

namespace boost {
  namespace text {
    struct repeated_string_view;
    constexpr repeated_string_view repeat(string_view, std::ptrdiff_t);
    bool operator==(repeated_string_view lhs, repeated_string_view rhs);
    bool operator!=(repeated_string_view lhs, repeated_string_view rhs);
    constexpr repeated_string_view::iterator begin(repeated_string_view rsv);
    constexpr repeated_string_view::iterator end(repeated_string_view rsv);
    constexpr repeated_string_view::iterator cbegin(repeated_string_view rsv);
    constexpr repeated_string_view::iterator cend(repeated_string_view rsv);
    constexpr repeated_string_view::reverse_iterator 
    rbegin(repeated_string_view rsv);
    constexpr repeated_string_view::reverse_iterator 
    rend(repeated_string_view rsv);
    constexpr repeated_string_view::reverse_iterator 
    crbegin(repeated_string_view rsv);
    constexpr repeated_string_view::reverse_iterator 
    crend(repeated_string_view rsv);
    int operator+(repeated_string_view lhs, repeated_string_view rhs);
    int operator+(string_view lhs, repeated_string_view rhs);
    int operator+(repeated_string_view lhs, string_view rhs);
  }
}
