struct grapheme {
  // types
  typedef utf_8_to_32_iterator< char const * > const_iterator;

  grapheme();

  // public member functions
  bool empty() const noexcept;
  int distance() const noexcept;
  const_iterator begin() const noexcept;
  const_iterator end() const noexcept;
  bool operator==(grapheme const &) const noexcept;
  bool operator!=(grapheme const &) const noexcept;

  // friend functions
  friend std::ostream & operator<<(std::ostream &, grapheme const &);

};

Description

An owning sequence of code points that comprise a grapheme.

grapheme public construct/copy/destruct

  1. grapheme();

    Default ctor. 

  2. template<typename CPIter> grapheme(CPIter first, CPIter last);

    Constructs *this from the code points [first, last).

    Requires:

    The code points in [first, last) comprise at most one grapheme.

    Requires:

    [first, last) is normalized FCC. 

  3. grapheme(uint32_t cp);

    Constructs *this from the code point cp. 

  4. template<typename CPIter> grapheme(utf32_view< CPIter > r);

    Constructs *this from r.

    Requires:

    The code points in r comprise at most one grapheme.

    Requires:

    The code points in [first, last) comprise at most one grapheme.

    Requires:

    [first, last) is normalized FCC.

grapheme public member functions

  1. bool empty() const noexcept;

    Returns true if *this contains no code points.

  4. const_iterator end() const noexcept;

  6. bool operator!=(grapheme const & other) const noexcept;

grapheme friend functions

    Returns the number of bytes controlled by g.
